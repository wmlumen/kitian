        // Inicializador del sistema de audio de J.A.R.V.I.S. (Web Audio API)
        let audioCtx = null;

        function initAudio() {
            if (!audioCtx) {
                audioCtx = new (window.AudioContext || window.webkitAudioContext)();
            }
        }

        // Sintetizador para efectos de sonido futuristas
        function playSound(type) {
            if (!sfxEnabled) return;
            try {
                initAudio();
                if (audioCtx.state === 'suspended') {
                    audioCtx.resume();
                }

                switch (type) {
                    case 'click':
                        let osc1 = audioCtx.createOscillator();
                        let gain1 = audioCtx.createGain();
                        osc1.type = 'sine';
                        osc1.frequency.setValueAtTime(1500, audioCtx.currentTime);
                        osc1.frequency.exponentialRampToValueAtTime(400, audioCtx.currentTime + 0.05);
                        gain1.gain.setValueAtTime(0.08, audioCtx.currentTime);
                        gain1.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.05);
                        osc1.connect(gain1);
                        gain1.connect(audioCtx.destination);
                        osc1.start();
                        osc1.stop(audioCtx.currentTime + 0.05);
                        break;
                    case 'alert':
                        let now = audioCtx.currentTime;
                        [2000, 2500].forEach((freq, idx) => {
                            let osc = audioCtx.createOscillator();
                            let gain = audioCtx.createGain();
                            osc.type = 'triangle';
                            osc.frequency.setValueAtTime(freq, now + idx * 0.06);
                            gain.gain.setValueAtTime(0.06, now + idx * 0.06);
                            gain.gain.exponentialRampToValueAtTime(0.001, now + idx * 0.06 + 0.05);
                            osc.connect(gain);
                            gain.connect(audioCtx.destination);
                            osc.start(now + idx * 0.06);
                            osc.stop(now + idx * 0.06 + 0.05);
                        });
                        break;
                    case 'thinking':
                        let osc3 = audioCtx.createOscillator();
                        let gain3 = audioCtx.createGain();
                        osc3.type = 'sine';
                        osc3.frequency.setValueAtTime(300, audioCtx.currentTime);
                        osc3.frequency.linearRampToValueAtTime(900, audioCtx.currentTime + 0.3);
                        
                        let filter = audioCtx.createBiquadFilter();
                        filter.type = 'lowpass';
                        filter.frequency.setValueAtTime(700, audioCtx.currentTime);

                        gain3.gain.setValueAtTime(0.03, audioCtx.currentTime);
                        gain3.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.35);

                        osc3.connect(filter);
                        filter.connect(gain3);
                        gain3.connect(audioCtx.destination);
                        osc3.start();
                        osc3.stop(audioCtx.currentTime + 0.35);
                        break;
                    case 'error':
                        let oscE = audioCtx.createOscillator();
                        let gainE = audioCtx.createGain();
                        oscE.type = 'sawtooth';
                        oscE.frequency.setValueAtTime(350, audioCtx.currentTime);
                        oscE.frequency.setValueAtTime(240, audioCtx.currentTime + 0.12);
                        gainE.gain.setValueAtTime(0.08, audioCtx.currentTime);
                        gainE.gain.exponentialRampToValueAtTime(0.001, audioCtx.currentTime + 0.25);
                        oscE.connect(gainE);
                        gainE.connect(audioCtx.destination);
                        oscE.start();
                        oscE.stop(audioCtx.currentTime + 0.25);
                        break;
                }
            } catch (e) {
                console.log("Audio awaiting user interaction:", e);
            }
        }

        // Convertidor Hexadecimal a degradados cian translúcidos de Stark
        function getShades(hexColor) {
            let r = parseInt(hexColor.slice(1,3), 16);
            let g = parseInt(hexColor.slice(3,5), 16);
            let b = parseInt(hexColor.slice(5,7), 16);
            
            return [
                `rgb(${r}, ${g}, ${b})`,
                `rgb(${Math.min(255, Math.floor(r*0.65))}, ${Math.min(255, Math.floor(g*0.65))}, ${Math.min(255, Math.floor(b*0.65))})`,
                `rgb(${Math.min(255, Math.floor(r*0.40))}, ${Math.min(255, Math.floor(g*0.40))}, ${Math.min(255, Math.floor(b*0.40))})`,
                `rgb(${Math.min(255, Math.floor(r*0.20))}, ${Math.min(255, Math.floor(g*0.20))}, ${Math.min(255, Math.floor(b*0.20))})`,
                `rgb(${Math.min(255, Math.floor(r*0.10))}, ${Math.min(255, Math.floor(g*0.10))}, ${Math.min(255, Math.floor(b*0.10))})`,
                `rgb(${Math.min(255, Math.floor(r*0.04))}, ${Math.min(255, Math.floor(g*0.04))}, ${Math.min(255, Math.floor(b*0.04))})`
            ];
        }

        // Cambio de estado global de J.A.R.V.I.S.
        function changeState(newState) {
            if (!COLORS[newState]) return;
            currentState = newState;
            targetColor = COLORS[newState][0];
            
            if (newState === 'Error') playSound('error');
            else if (newState === 'Pensando') playSound('thinking');
            else playSound('click');

            appendLog(`[SISTEMA]: Estado de J.A.R.V.I.S. modificado a -> ${newState.toUpperCase()}`);

            document.querySelectorAll('.state-btn').forEach(btn => {
                const stateType = btn.id.split('-')[1];
                if (stateType === newState) {
                    btn.classList.add('bg-cyan-500/10', 'text-cyan-400');
                } else {
                    btn.classList.remove('bg-cyan-500/10', 'text-cyan-400');
                }
            });

            const badge = document.getElementById('footer-state-badge');
            
            if (newState === 'Activo') { 
                badge.innerText = "STANDBY";
                badge.className = "px-1.5 py-0.5 text-black bg-[#00f0ff] rounded font-bold"; 
            }
            else if (newState === 'Escuchando') { 
                badge.innerText = "ESCUCHANDO";
                badge.className = "px-1.5 py-0.5 text-black bg-cyan-400 rounded font-bold";

                badge.innerText = "PROCESANDO";
                badge.className = "px-1.5 py-0.5 text-white bg-purple-600 rounded font-bold"; 
            }
            else if (newState === 'Hablando') { 
                badge.innerText = "HABLANDO";
                badge.className = "px-1.5 py-0.5 text-black bg-amber-400 rounded font-bold"; 
            }
            else if (newState === 'Offline') { 
                badge.innerText = "APAGADO";
                badge.className = "px-1.5 py-0.5 text-white bg-zinc-600 rounded font-bold"; 
            }
            else if (newState === 'Error') { 
                badge.innerText = "ANOMALÍA";
                badge.className = "px-1.5 py-0.5 text-white bg-red-600 rounded font-bold"; 
            }

            const subtitle = document.getElementById('subtitles-display');

            const container = document.getElementById('terminal-logs');

        function submitConsoleCommand() {
            const input = document.getElementById('console-input');
            const rawVal = input.value.trim();
            if (!rawVal) return;
            
            appendLog(`<span class="text-cyan-300">&gt; ${rawVal}</span>`);
            input.value = "";
            playSound('click');

            const val = rawVal.toUpperCase();

            if (val === 'HELP' || val === '/HELP') {
                appendLog("[ASISTENTE]: Comandos disponibles: VERONICA, SISTEMA, COMPILAR, ANALIZAR, LIMPIAR, INICIAR [ACTIVO/ESCUCHANDO/PENSANDO/HABLANDO/OFFLINE/ERROR]");
            } else if (val === 'LIMPIAR' || val === '/CLEAR' || val === 'CLEAR') {
                document.getElementById('terminal-logs').innerHTML = "";
                appendLog("[SISTEMA]: Se ha liberado la memoria intermedia de logs.");
            } else if (val === 'SISTEMA' || val === 'SYS') {
                appendLog("[DIAGNÓSTICO]: Estado de Verónica: Listo. Escudos térmicos al 100%. Protocolo de combate stand-by.");
            } else if (val === 'VERONICA' || val === 'VERÓNICA') {
                changeState('Pensando');
                appendLog("[ÓRDEN]: Solicitando despliegue de armadura pesada Verónica desde órbita...");
                setTimeout(() => {
                    changeState('Hablando');
                    document.getElementById('subtitles-display').innerText = "J.A.R.V.I.S.: \"Verónica se encuentra en posición orbital de reingreso, Señor.\"";
                    setTimeout(() => changeState('Activo'), 3000);
                }, 2000);
            } else if (val === 'ANALIZAR' || val === 'ANALYZE') {
                changeState('Pensando');
                setTimeout(() => {
                    changeState('Activo');
                    appendLog("[RESULTADOS]: Calibración del Reactor X20 finalizada. Pureza del Paladio óptima.");
                }, 2500);
            } else if (val === 'COMPILAR' || val === 'OPTIMIZE') {
                changeState('Pensando');
                setTimeout(() => {
                    changeState('Hablando');
                    appendLog("[SISTEMA]: Limpieza de memoria temporal completa.");
                    setTimeout(() => changeState('Activo'), 2000);
                }, 3000);
            } else if (val.startsWith('INICIAR ')) {
                const parts = val.split(' ');
                const reqState = parts[1].charAt(0) + parts[1].slice(1).toLowerCase();
                if (COLORS[reqState]) {
                    changeState(reqState);
                } else {
                    appendLog(`[ERROR]: Parámetro de estado no válido.`);
                }
            } else {
                appendLog(`[J.A.R.V.I.S.]: Procesando palabra clave en servidores Stark...`);
                changeState('Pensando');
                setTimeout(() => {
                    changeState('Hablando');
                    document.getElementById('subtitles-display').innerText = `J.A.R.V.I.S.: "He ejecutado el subproceso táctico para '${rawVal.toUpperCase()}'. Sin complicaciones, Señor."`;
                    setTimeout(() => changeState('Activo'), 2500);
                }, 1800);
            }
        }

        // 1. Gráfico de Analítica Predictiva
        function drawPredictiveGraph() {
            pCtx.clearRect(0, 0, predictiveCanvas.width, predictiveCanvas.height);
            pCtx.strokeStyle = "rgba(0, 240, 255, 0.6)";
            pCtx.lineWidth = 1.2;
            pCtx.beginPath();
            const stepP = predictiveCanvas.width / (predictiveHistory.length - 1);
            for(let i = 0; i < predictiveHistory.length; i++) {
                const x = i * stepP;
                if (frameCount % 8 === 0) {
                    predictiveHistory[i] = 40 + Math.sin(frameCount * 0.02 + i * 0.5) * 15 + Math.random() * 5;
                }
                const y = predictiveCanvas.height - (predictiveHistory[i] / 100 * (predictiveCanvas.height - 20)) - 10;
                if (i === 0) pCtx.moveTo(x, y);
                else pCtx.lineTo(x, y);
            }
            pCtx.stroke();
        }

        // 2. Sinapsis de Red Neuronal (3D Brain Simulation)
        let synapseNodes = [];
        function initSynapses() {
            synapseNodes = [];
            const count = 24;
            for(let i=0; i<count; i++) {
                synapseNodes.push({
                    x: Math.random() * 200 + 20,
                    y: Math.random() * 80 + 15,
                    vx: (Math.random() - 0.5) * 0.4,
                    vy: (Math.random() - 0.5) * 0.4,
                    r: Math.random() * 2 + 1
                });
            }
        }

        function drawSynapses() {
            sCtx.clearRect(0, 0, synapseCanvas.width, synapseCanvas.height);
            sCtx.fillStyle = "rgba(0, 240, 255, 0.7)";
            sCtx.strokeStyle = "rgba(0, 240, 255, 0.15)";
            sCtx.lineWidth = 0.5;

            // Update & Draw nodes
            synapseNodes.forEach((node, idx) => {
                node.x += node.vx;
                node.y += node.vy;

                // Bounce
                if(node.x < 0 || node.x > synapseCanvas.width) node.vx *= -1;
                if(node.y < 0 || node.y > synapseCanvas.height) node.vy *= -1;

                sCtx.beginPath();
                sCtx.arc(node.x, node.y, node.r, 0, Math.PI * 2);
                sCtx.fill();

                // Draw connections
                for(let j = idx + 1; j < synapseNodes.length; j++) {
                    const other = synapseNodes[j];
                    const dist = Math.hypot(node.x - other.x, node.y - other.y);
                    if(dist < 45) {
                        sCtx.beginPath();
                        sCtx.moveTo(node.x, node.y);
                        sCtx.lineTo(other.x, other.y);
                        sCtx.stroke();
                    }
                }
            });
        }

        // 3. Globo 3D Wireframe Rotatorio
        let globeAngleY = 0;
        function drawGlobe() {
            gCtx.clearRect(0, 0, globeCanvas.width, globeCanvas.height);
            const cx = globeCanvas.width / 2;
            const cy = globeCanvas.height / 2;
            const r = 40;

            gCtx.strokeStyle = "rgba(0, 240, 255, 0.15)";
            gCtx.lineWidth = 0.5;

            globeAngleY += 0.007;

            // Draw parallels (Latitudes)
            for(let lat = -Math.PI/2; lat <= Math.PI/2; lat += Math.PI/6) {
                const pr = r * Math.cos(lat);
                const py = cy + r * Math.sin(lat);
                
                gCtx.beginPath();
                gCtx.ellipse(cx, py, pr, pr * 0.25, 0, 0, Math.PI * 2);
                gCtx.stroke();
            }

            // Draw meridians (Longitudes)
            for(let lon = 0; lon < Math.PI * 2; lon += Math.PI/6) {
                const effectiveLon = lon + globeAngleY;
                const rx = r * Math.cos(effectiveLon);
                
                gCtx.strokeStyle = Math.sin(effectiveLon) > 0 ? "rgba(0, 240, 255, 0.25)" : "rgba(0, 240, 255, 0.08)";
                gCtx.beginPath();
                
                // Draw longitude curves as ellipses
                gCtx.ellipse(cx, cy, Math.abs(rx), r, 0, 0, Math.PI * 2);
                gCtx.stroke();
            }

            // Outer glow ring
            gCtx.strokeStyle = "rgba(0, 240, 255, 0.4)";
            gCtx.lineWidth = 1;
            gCtx.beginPath();
            gCtx.arc(cx, cy, r + 4, 0, Math.PI*2);
            gCtx.stroke();
        }

        // 4. Caza Estelar 3D (Spaceship Wireframe Projection)
        let ssAngleY = 0;
        let ssAngleX = 0.2;
        // Vertices 3D
        const ssVertices = [
            {x: 0, y: 0, z: 35},    // 0: Nose
            {x: 12, y: 0, z: -15},  // 1: Right wing-join
            {x: -12, y: 0, z: -15}, // 2: Left wing-join
            {x: 0, y: 6, z: -15},   // 3: Tail-join top
            {x: 0, y: -4, z: -15},  // 4: Engine body bottom
            {x: 30, y: -2, z: -8},  // 5: Wing tip Right
            {x: -30, y: -2, z: -8}, // 6: Wing tip Left
            {x: 0, y: 15, z: -25}   // 7: Vertical stabilizer tip
        ];
        // Conexiones de aristas (Edges)
        const ssEdges = [
            [0, 1], [0, 2], [0, 3], [0, 4],
            [1, 5], [5, 2], [2, 6], [6, 1],
            [1, 3], [2, 3], [1, 4], [2, 4],
            [3, 7], [7, 4], [7, 0]
        ];

        function drawStarship() {
            ssCtx.clearRect(0, 0, starshipCanvas.width, starshipCanvas.height);
            const cx = starshipCanvas.width / 2;
            const cy = starshipCanvas.height / 2;
            const scale = 1.3;

            ssAngleY += 0.01;
            ssAngleX = 0.2 + 0.1 * Math.sin(frameCount * 0.02);

            // Project vertices
            const projected = ssVertices.map(v => {
                // Rotation Y
                let x1 = v.x * Math.cos(ssAngleY) - v.z * Math.sin(ssAngleY);
                let z1 = v.x * Math.sin(ssAngleY) + v.z * Math.cos(ssAngleY);
                // Rotation X
                let y2 = v.y * Math.cos(ssAngleX) - z1 * Math.sin(ssAngleX);
                let z2 = v.y * Math.sin(ssAngleX) + z1 * Math.cos(ssAngleX);

                // Ortographic projection
                return {
                    x: cx + x1 * scale,
                    y: cy - y2 * scale
                };
            });

            // Draw edges
            ssCtx.strokeStyle = "rgba(0, 240, 255, 0.45)";
            ssCtx.lineWidth = 0.8;
            ssEdges.forEach(edge => {
                const p1 = projected[edge[0]];
                const p2 = projected[edge[1]];
                ssCtx.beginPath();
                ssCtx.moveTo(p1.x, p1.y);
                ssCtx.lineTo(p2.x, p2.y);
                ssCtx.stroke();
            });

            // Draw points
            ssCtx.fillStyle = "#00ff88";
            projected.forEach(p => {
                ssCtx.beginPath();
                ssCtx.arc(p.x, p.y, 1.5, 0, Math.PI * 2);
                ssCtx.fill();
            });
        }

        // 5. Radar de amenazas circular
        let radarSweepAngle = 0;
        function drawRadar() {
            radCtx.clearRect(0, 0, radarCanvas.width, radarCanvas.height);
            const cx = radarCanvas.width / 2;
            const cy = radarCanvas.height / 2;
            const r = 45;

            // Dibujar cuadrícula de círculos de alcance
            radCtx.strokeStyle = "rgba(0, 240, 255, 0.1)";
            radCtx.lineWidth = 0.5;
            for(let i=1; i<=3; i++) {
                radCtx.beginPath();
                radCtx.arc(cx, cy, r * (i/3), 0, Math.PI * 2);
                radCtx.stroke();
            }

            // Dibujar cruz del radar
            radCtx.beginPath();
            radCtx.moveTo(cx - r, cy); radCtx.lineTo(cx + r, cy);
            radCtx.moveTo(cx, cy - r); radCtx.lineTo(cx, cy + r);
            radCtx.stroke();

            // Dibujar barrido de luz
            radarSweepAngle += 0.02;
            radCtx.strokeStyle = "rgba(0, 240, 255, 0.4)";
            radCtx.lineWidth = 1;
            radCtx.beginPath();
            radCtx.moveTo(cx, cy);
            radCtx.lineTo(cx + r * Math.cos(radarSweepAngle), cy + r * Math.sin(radarSweepAngle));
            radCtx.stroke();

            // Dibujar objetivos simulados (Targets pulsing)
            radCtx.fillStyle = "rgba(255, 42, 0, " + (0.3 + 0.7 * Math.abs(Math.sin(frameCount * 0.05))) + ")";
            const targetPos = [
                {x: cx + r * 0.5, y: cy - r * 0.3},
                {x: cx - r * 0.6, y: cy + r * 0.4}
            ];
            targetPos.forEach(t => {
                radCtx.beginPath();
                radCtx.arc(t.x, t.y, 2, 0, Math.PI*2);
                radCtx.fill();
            });
        }

        // --- ARQUITECTURA DEL REACTOR ARC CENTRAL (CANVAS) ---
        let arcRotationAngle = 0;
        function drawArcReactor() {
            const W = arcReactorCanvas.width / window.devicePixelRatio;
            const H = arcReactorCanvas.height / window.devicePixelRatio;
            const CX = W / 2;
            const CY = H / 2;

            rCtx.clearRect(0, 0, W, H);
            arcRotationAngle += 0.02 * animationSpeed;

            const shades = getShades(targetColor);
            const activeColor = shades[0];
            const R = 54 + 4 * Math.sin(frameCount * 0.04);

            // Anillos concéntricos asimétricos segmentados
            for (let i = 12; i > 0; i--) {
                const gr = R + 5 + i * 11;
                const shadeIdx = Math.min(i, 5);
                rCtx.strokeStyle = shades[shadeIdx];
                rCtx.lineWidth = i > 6 ? 0.4 : 0.8;
                rCtx.beginPath();
                
                // Segments
                if (i % 3 === 0) {
                    rCtx.setLineDash([12, 18]);
                } else if (i % 2 === 0) {
                    rCtx.setLineDash([35, 12]);
                } else {
                    rCtx.setLineDash([]);
                }
                
                rCtx.arc(CX, CY, gr, arcRotationAngle * (i * 0.015), Math.PI * 2 + arcRotationAngle * (i * 0.015));
                rCtx.stroke();
                rCtx.setLineDash([]);
            }

            // Reguladores magnéticos (Líneas radiales de telemetría)
            rCtx.strokeStyle = shades[3];
            rCtx.lineWidth = 0.5;
            for (let j = 0; j < 48; j++) {
                const a = (j * Math.PI / 24) + arcRotationAngle * 0.04;
                const x1 = CX + (R + 42) * Math.cos(a);

El "Mapa Mundi" que giraNombre técnico: Globo de Estructura de Alambre 3D con Proyección Ortográfica (3D Wireframe Globe with Orthographic Projection).Cómo funciona en programación:Para dibujar este planeta en una pantalla plana (2D) sin usar motores pesados como Unity o Three.js, usamos trigonometría esférica.Se definen líneas de latitud (paralelos) y longitud (merididanos) mediante ángulos de rotación.Se utiliza una Matriz de Rotación para girar esos puntos virtuales en el espacio de 3 dimensiones usando una velocidad constante ($\theta$).Para proyectar el punto tridimensional $(x, y, z)$ en la pantalla bidimensional $(x, y)$, se utiliza una Proyección Ortográfica. Es decir, simplemente ignoramos o aplanamos la profundidad ($z$) para dibujar círculos y elipses que se estiran y encogen simulando una esfera en 3D:$$x_{\text{pantalla}} = CX + R \cdot \cos(\text{longitud} + \theta)$$$$y_{\text{pantalla}} = CY + R \cdot \sin(\text{latitud})$$2. La "Red Neuronal"Nombre técnico: Efecto Constelación o Grafo Dinámico de Partículas (Constellation Effect / Dynamic Particle Node-Link Graph).Cómo funciona en programación:Este sistema consta de dos algoritmos simples pero muy estéticos:Motor de física de partículas básico: Cada "nodo" (punto) es una partícula independiente que tiene una posición $(x, y)$ y un vector de velocidad $(v_x, v_y)$. En cada fotograma, sumamos la velocidad a la posición y aplicamos una colisión elástica con los bordes (hacer que "reboten" multiplicando su velocidad por $-1$ si tocan los límites del panel).Cálculo de Distancia Euclidiana (Enlace de Grafo): En cada ciclo de dibujado, un bucle compara la distancia entre cada par de puntos usando el teorema de Pitágoras:$$\text{distancia} = \sqrt{(x_2 - x_1)^2 + (y_2 - y_1)^2}$$Si la distancia es menor a un umbral (por ejemplo, $45$ píxeles), el programa dibuja una línea (canvas.create_line) entre ellos. Si están muy lejos, la línea no se dibuja. Esto crea el efecto orgánico de conexiones neuronales que se crean y destruyen.3. El "Círculo" central que se mueve (Reactor Arc)Nombre técnico: Renderizado de Vectores en Coordenadas Polares y Arcos de Barrido (Polar Coordinates Vector Rendering & Sweep Arcs).Cómo funciona en programación:En lugar de usar las coordenadas de cuadrícula normales (Cartesianas: $X, Y$), las animaciones circulares se programan usando Coordenadas Polares, las cuales se definen por un radio ($R$) y un ángulo ($\theta$).Para dibujar los arcos giratorios segmentados, se utilizan comandos nativos del Canvas como create_arc, alterando sus ángulos de inicio y extensión en cada fotograma.Para colocar las palabras (ANALYZE, ADAPT, etc.) o las líneas radiales que rotan alrededor del círculo, el código realiza una conversión de coordenadas polares a cartesianas en cada ciclo de animación para saber exactamente en qué píxel de la pantalla debe pintar el elemento:$$X = CX + R \cdot \cos(\theta)$$$$Y = CY + R \cdot \sin(\theta)$$4.
